
# Load an image into the docker environment.
docker load < myimage.tar

# Run a container interactively with bash from 
# an image with a given tag. Also "-t tip".
docker run -i -t e05477ddd807 /bin/bash

# In tip dir with Dockerfile located here.
sudo docker build -t tip .
sudo docker run -it tip /bin/bash

# Build an image from a docker file with the ability to import 
# files/dirs from .
docker build -t tip -f container/docker/centos/Dockerfile .

# Export an image as a gzipped tarball.
docker save tip:latest | gzip > tip_docker_image.tar.gz
